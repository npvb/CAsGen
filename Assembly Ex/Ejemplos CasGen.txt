int gcd(int a, int b)
{
   if (b == 0) 
   {
      return a;
   }
   else 
   {
      return gcd(b, a % b);      
   }
}

int fib(int x) 
{
    if(x <= 1)
    {
	return x;
    }else
    {
        return fib(x-1) + fib(x-2);

void bubbleSort()
{

int a[50];
int n;
int i;
int j;
int temp;

for(i=1;i<n;++i)
{
        for(j=0;j<(n-i);++j)
	{
            if(a[j]>a[j+1])
            {
		int b = j+1
                temp=a[j];
                a[j]=a[b];
                a[b]=temp;
            }
    	}
}
}
